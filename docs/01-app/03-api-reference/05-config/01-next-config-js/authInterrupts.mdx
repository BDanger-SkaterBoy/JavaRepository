The `authInterrupts` configuration option allows you to use [`unstable_forbidden`](/docs/app/api-reference/functions/unstable_forbidden) and [`unstable_unauthorized`](/docs/app/api-reference/functions/unstable_unauthorized) in your application. While these functions are experimental, you must enable the `authInterrupts` option in your `next.config.js` file to use them:

```ts filename="next.config.ts" switcher
import type { NextConfig } from 'next'

const nextConfig: NextConfig = {
  experimental: {
    authInterrupts: true,
  },
}

export default nextConfig
```

```js filename="next.config.js" switcher
module.exports = {
  experimental: {
    authInterrupts: true,
  },
}
```
